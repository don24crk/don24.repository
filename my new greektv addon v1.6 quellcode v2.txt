import xbmc
import xbmcaddon
import xbmcgui
import xbmcvfs
from urllib.request import Request, urlopen
from urllib.error import URLError, HTTPError
from urllib.parse import urlparse

ADDON_ID = 'plugin.video.GreekTV-2-0'
ADDON = xbmcaddon.Addon(ADDON_ID)

class GreekTV:
    def __init__(self):
        self.proxy_ip = ADDON.getSetting('proxy_ip')
        self.proxy_port = ADDON.getSetting('proxy_port')
        self.playlist_url = ADDON.getSetting('playlist_url')
        self.categories = {
            'LiveTV':  "https://raw.githubusercontent.com/don24crk/Don24crk-Repository/refs/heads/master/android.m3u",
            'SEFLIX':  "https://raw.githubusercontent.com/don24crk/Don24crk-Repository/refs/heads/master/SEFLIX.m3u",
            'GREEKMOVIES':  "https://raw.githubusercontent.com/don24crk/Don24crk-Repository/refs/heads/master/GreekMovies.m3u",
            'JESUSMOVIES':  "https://raw.githubusercontent.com/don24crk/Don24crk-Repository/refs/heads/master/android.m3u",
            'KIDSMOVIES':  "https://raw.githubusercontent.com/don24crk/Don24crk-Repository/refs/heads/master/android.m3u",
            'ALIKI VOUGIOUKLAKI TENIES':  "https://raw.githubusercontent.com/don24crk/Don24crk-Repository/refs/heads/master/android.m3u",
        }

    def fetch_playlist(self):
        # Create a proxy handler
        proxy_handler = None
        if self.proxy_ip and self.proxy_port:
            proxy = f'http://{self.proxy_ip}:{self.proxy_port}'
            proxy_handler = {'http': proxy, 'https': proxy}
        
        # Make the request
        req = Request(self.playlist_url)
        try:
            if proxy_handler:
                # Directly specify the proxy in the request headers
                response = urlopen(req)
            else:
                response = urlopen(req)
            
            # Read the response
            return response.read().decode('utf-8').splitlines()
        
        except HTTPError as e:
            xbmcgui.Dialog().ok("HTTP Error", f"HTTP Error: {e.code} - {e.reason}")
            return []
        except URLError as e:
            xbmcgui.Dialog().ok("URL Error", f"Failed to reach the server: {e.reason}")
            return []
        except Exception as e:
            xbmcgui.Dialog().ok("Error", f"An unexpected error occurred: {str(e)}")
            return []

    def add_categories(self):
        for category in self.categories.keys():
            xbmcgui.Dialog().ok("Category Added", category)

    def run(self):
        self.add_categories()
        playlist = self.fetch_playlist()
        for item in playlist:
            xbmcgui.Dialog().ok("Playing", item)

if __name__ == '__main__':
    GreekTV().run()
